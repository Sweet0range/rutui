        def create
                obj = []
    (0..(@radius*2)).each do |x|
      (0..(@radius*2)).each do |y|
                                obj[y] = [] if obj[y].nil?
        obj[y][x] = distance_from_center(x,y).round == @radius ? @pixel : @fill
      end
    end
                @obj = obj
        end
