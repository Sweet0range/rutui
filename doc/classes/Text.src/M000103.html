        def create
                if @do_rainbow
                        rainbow = 0
                end
                @width = @text.size
                @obj = []
                tmp = []
                @text.split("").each do |t|
                        if @do_rainbow
                                tmp << Pixel.new(@@rainbow[rainbow],@bg,t)
                                rainbow += 1
                                rainbow = 0 if rainbow >= @@rainbow.size
                        else
                                tmp << Pixel.new(@fg,@bg,t)
                        end
                end
                @obj << tmp
        end
